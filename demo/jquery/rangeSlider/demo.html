滑块——slider
rangeSlider：官网
价格区间：
            var price = $("#price");
            $("#test").ionRangeSlider({
                min: 0,           // min 最小值
                max: 5000,        // max 最大值
                from: 1000,       // from 起始点
                to: 4000,         // to 结束点
                type: 'double',   // 滑块条类型
                step: 1,          // 滑块移动步长
                prefix: "¥",      // 在数值前添加字符文本
                prettify: false,  // 数值是不是采用空格分隔
                hasGrid: true,    // 添加样式属性
                // 初始化时执行
                onLoad: function (obj) {        
                    price.html(obj.fromNumber+"元 至 "+obj.toNumber+"元");
                },
                // 滑块移动时执行
                onChange: function (obj) {      
                    price.html(obj.fromNumber+"元 至"+obj.toNumber+"元");
                }
            });
