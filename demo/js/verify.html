正则表达式基础介绍
            /* 常用符号 */

            // ^匹配字符串的开始
            // $匹配字符串的结束
            // 例如：匹配必须以1开始和数字结束的字符串。
            console.log(/^1\d{10}$/.test("13699999999"));
            结果：true

            // . 匹配任意字符除换行符
            // \w 匹配字母数字下划线 等价于[a-zA-Z0-9_]
            // \W 匹配除字母数字下划线以外的字符 等价于[^a-zA-Z0-9_]
            // \d 匹配数字 等价于[0-9]
            // \D 匹配非数字 等价于[^0-9]
            // \s 匹配空格
            // \S 匹配非空格
            // 例如：匹配6-20个由字母数字或下划线的字符串
            console.log(/^\w{6,20}$/.test("abcdef"));
            结果：true
            // 例如：匹配6-20个由字母数字或下划线的字符串，必须以字母开始
            console.log(/^[a-zA-Z]\w{5,19}$/.test("abcedf"));
            结果：true
            // 例如：过滤字符串两边的空格
            console.log("      ab   ".replace(/^(\s*)(\S*)(\s*)$/,"$2"));
            结果：ab
            console.log("      ab   ".replace(/^(\s*)|(\s*)$/g,""));
            结果：ab


            // * 代表零次或多次 等价于{0,}
            // + 代表一次或多次 等价于{1,}
            // ? 代表零次或一次 等价于{0,1}
            // {n} 代表重复n次
            // {n,} 代表至少重复n次
            // {n,m} 代表重复n次到m次
            // 例如：检查手机号码
            console.log(/^1[3|4|5|8]\d{9}/.test("123456789"));
            结果：false
            // 例如：检查座机电话号码
            console.log(/^(\d{3}-\d{8})|(\d{4}-\d{7})$/.test("010-12345678"));
            结果：true


            // () 标记字表达式的开始和结束
            // [] 包括
            // [^] 不包括
            // 例如：检查匹配100001-9999999999的正数
            console.log(/^(?!100000)[1-9]\d{5,9}$/.test("100000"));
            结果：false
            // 例如：检查图片的格式
            console.log(/[jpg|gif|png]/ig.test("test.png"));
            结果：true
            // 例如：除abc字母排列组合的所有字符，空，a,b,c,ab,ba,bc,cb,ca,ac,abc,bac...
            console.log(/[^abc]/g.test(""));
            结果：false
