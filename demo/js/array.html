一、创建数组 
      // 创建空数组
      var arr1 = new Array();
      var arr2 = [];
      // 创建指定长度数组
      var arr3 = new Array(6);
      // 创建指定元素数组
      var arr4 = new Array("a","b","c");
      var arr5 = ["a","b","c"];
二、属性介绍

1、length

      var arr1 = new Array();
      console.log(arr1.length);
      结果：0 
      var arr3 = new Array(6);
      console.log(arr3.length);
      结果：6
三、方法介绍

1、concat

      // concat:将两个或多个数组链接成一个新数组,返回新数组。
      var arr1 = ["a","b","c"];
      var arr2 = ["e","f"];
      var elem = "g";
      console.log(arr1.concat(arr2,elem));
      结果：["a", "b", "c", "e", "f", "g"] 
      console.log(arr1);
      结果：["a","b","c"]

2、join

      // join:把数组元素由分隔符连接成字符串，返回字符串。
      var arr1 = ["a","bb","abc"];
      console.log(arr1.join("---"));
      结果：a---bb---abc

3、pop

      // pop:删除数组最后一个元素，返回该元素
      var arr1 = ["a","bb","abc"];
      console.log(arr1.pop());
      结果：abc
      console.log(arr1);
      结果：["a", "bb"]

4、shift

      // shift删除数组的第一元素，返回该元素
      var arr1 = ["a","bb","abc"];
      console.log(arr1.shift());
      结果：a
      console.log(arr1);
      结果：["bb", "abc"]

5、push

      // push向数组末尾添加一个或多个元素，返回数组长度
      var arr1 = ["a","bb","abc"];
      console.log(arr1.push("dd","df"));
      结果：5
      console.log(arr1);
      结果：["a", "bb", "abc", "dd", "df"]
